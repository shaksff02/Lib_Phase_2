üìù Creating Book Form

Create forms.py File

Create new file: myapp/forms.py

from django import forms from .models import Book class BookForm(forms.ModelForm): class Meta: model = Book fields = ['title', 'author', 'isbn', 'quantity'] widgets = { 'title': forms.TextInput( attrs={ 'class': 'form-control', 'placeholder': 'Book Title' } ), 'author': forms.TextInput( attrs={ 'class': 'form-control', 'placeholder': 'Author Name' } ), 'isbn': forms.TextInput( attrs={ 'class': 'form-control', 'placeholder': '13-digit ISBN' } ), 'quantity': forms.NumberInput( attrs={ 'class': 'form-control', 'min': '1' } ), }
üìñ What is ModelForm?

ModelForm automatically creates a form based on your model fields. It saves you from manually defining each field. The widgets customize how fields appear in HTML.

üëÅÔ∏è CRUD Views

1. Create Book View

def create_book(request): if request.method == 'POST': form = BookForm(request.POST) if form.is_valid(): form.save() return redirect('myapp:book_list') else: form = BookForm() context = {'form': form} return render(request, 'myapp/create_book.html', context)

2. Book Detail View

def book_detail(request, pk): try: book = Book.objects.get(pk=pk) except Book.DoesNotExist: return render(request, 'myapp/404.html', {'message': 'Book not found'}) context = {'book': book} return render(request, 'myapp/book_detail.html', context)

3. Edit Book View

def edit_book(request, pk): try: book = Book.objects.get(pk=pk) except Book.DoesNotExist: return render(request, 'myapp/404.html') if request.method == 'POST': form = BookForm(request.POST, instance=book) if form.is_valid(): form.save() return redirect('myapp:book_detail', pk=book.pk) else: form = BookForm(instance=book) context = {'form': form, 'book': book} return render(request, 'myapp/edit_book.html', context)

4. Delete Book View

def delete_book(request, pk): try: book = Book.objects.get(pk=pk) except Book.DoesNotExist: return render(request, 'myapp/404.html') if request.method == 'POST': book.delete() return redirect('myapp:book_list') context = {'book': book} return render(request, 'myapp/delete_book.html', context)
‚ÑπÔ∏è Note

Add these views to myapp/views.py. Don't forget to import BookForm and redirect at the top!

üîó URL Routes for CRUD

Update myapp/urls.py

from django.urls import path from . import views app_name = 'myapp' urlpatterns = [ path('books/', views.book_list, name='book_list'), path('books/create/', views.create_book, name='create_book'), path('books/<int:pk>/', views.book_detail, name='book_detail'), path('books/<int:pk>/edit/', views.edit_book, name='edit_book'), path('books/<int:pk>/delete/', views.delete_book, name='delete_book'), ]
URL Pattern View Purpose
/books/ book_list List all books
/books/create/ create_book Create new book
/books/5/ book_detail View book with ID 5
/books/5/edit/ edit_book Edit book with ID 5
/books/5/delete/ delete_book Delete book with ID 5
